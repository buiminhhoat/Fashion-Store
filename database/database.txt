-- Xóa database nếu tồn tại và tạo lại
DROP DATABASE IF EXISTS fashionstore;
CREATE DATABASE fashionstore;
USE fashionstore;

-- Bảng User
CREATE TABLE Users (
    UserID INT PRIMARY KEY AUTO_INCREMENT,
    FullName VARCHAR(255) NOT NULL,
    Email VARCHAR(255) UNIQUE NOT NULL,
    PhoneNumber VARCHAR(20),
    Gender VARCHAR(10),
    DateBirthday DATE,
    AvatarPath VARCHAR(255)
);

-- Bảng Address
CREATE TABLE Addresses (
    AddressID INT PRIMARY KEY AUTO_INCREMENT,
    UserID INT,
    RecipientName VARCHAR(255) NOT NULL,
    RecipientPhone VARCHAR(20),
    RecipientCity VARCHAR(255),
    RecipientDistrict VARCHAR(255),
    RecipientWard VARCHAR(255),
    AddressDetails VARCHAR(255),
    FOREIGN KEY (UserID) REFERENCES Users(UserID)
);

-- Bảng Orders
CREATE TABLE Orders (
    OrderID INT PRIMARY KEY AUTO_INCREMENT,
    OrderDate DATE,
    TotalAmount DECIMAL(10, 2),
    OrderStatus VARCHAR(20),
    UserID INT,
    FOREIGN KEY (UserID) REFERENCES Users(UserID)
);

-- Bảng ProductCategory
CREATE TABLE ProductCategory (
    CategoryID INT PRIMARY KEY AUTO_INCREMENT,
    CategoryName VARCHAR(255) NOT NULL
);

-- Bảng Products
CREATE TABLE Products (
    ProductID INT PRIMARY KEY AUTO_INCREMENT,
    ProductName VARCHAR(255) NOT NULL,
    ProductPrice DECIMAL(10, 2),
    UnitsSold INT,
    Description TEXT,
    CategoryID INT,
    FOREIGN KEY (CategoryID) REFERENCES ProductCategory(CategoryID)
);

-- Bảng OrderDetails
CREATE TABLE OrderDetails (
    OrderDetailID INT PRIMARY KEY AUTO_INCREMENT,
    OrderID INT,
    ProductID INT,
    ProductPrice DECIMAL(10, 2),
    Quantity INT,
    TotalPrice DECIMAL(10, 2),
    FOREIGN KEY (OrderID) REFERENCES Orders(OrderID),
    FOREIGN KEY (ProductID) REFERENCES Products(ProductID)
);

-- Bảng ProductSizes
CREATE TABLE ProductSizes (
    SizeID INT PRIMARY KEY AUTO_INCREMENT,
    ProductID INT,
    SizeName VARCHAR(20),
    FOREIGN KEY (ProductID) REFERENCES Products(ProductID)
);

-- Bảng ProductImages
CREATE TABLE ProductImages (
    ImageID INT PRIMARY KEY AUTO_INCREMENT,
    ProductID INT,
    ImagePath VARCHAR(255),
    FOREIGN KEY (ProductID) REFERENCES Products(ProductID)
);

-- Bảng ProductColors
CREATE TABLE ProductColors (
    ColorID INT PRIMARY KEY AUTO_INCREMENT,
    ProductID INT,
    ColorName VARCHAR(20),
    ImagePath VARCHAR(255),
    INDEX idx_ColorID (ColorID),
    FOREIGN KEY (ProductID) REFERENCES Products(ProductID)
);

-- Bảng Inventory
CREATE TABLE Inventory (
    InventoryID INT PRIMARY KEY AUTO_INCREMENT,
    ProductID INT,
    QuantityOnHand INT,
    SizeID INT,
    ColorID INT,
    FOREIGN KEY (ProductID) REFERENCES Products(ProductID),
    FOREIGN KEY (SizeID) REFERENCES ProductSizes(SizeID),
    FOREIGN KEY (ColorID) REFERENCES ProductColors(ColorID)
);

-- Tạo mối quan hệ giữa Product và ProductCategory
ALTER TABLE Products
ADD CONSTRAINT FK_Products_Category FOREIGN KEY (CategoryID) REFERENCES ProductCategory(CategoryID);
